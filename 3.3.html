Person → Student & Teacher (ES6 classes)

This demo shows a base Person class and two subclasses (Student, Teacher) that extend it. Each subclass extends or overrides the displayInfo method to include its own properties.

Source (important parts)

// Base class
class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  // method to return a basic description
  displayInfo() {
    return `Name: ${this.name} | Age: ${this.age}`;
  }
}

// Student extends Person
class Student extends Person {
  constructor(name, age, course) {
    super(name, age);
    this.course = course;
  }

  // override to add course information
  displayInfo() {
    // call the parent method and extend
    return `${super.displayInfo()} | Course: ${this.course}`;
  }
}

// Teacher extends Person
class Teacher extends Person {
  constructor(name, age, subject) {
    super(name, age);
    this.subject = subject;
  }

  // override to add subject/department information
  displayInfo() {
    return `${super.displayInfo()} | Subject: ${this.subject}`;
  }
}

// usage: create instances and call displayInfo()
const s = new Student('Aisha', 20, 'Computer Science');
const t = new Teacher('Mr. Rao', 42, 'Mathematics');

console.log(s.displayInfo());
console.log(t.displayInfo());
